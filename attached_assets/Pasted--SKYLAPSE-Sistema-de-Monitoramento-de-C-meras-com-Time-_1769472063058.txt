# SKYLAPSE - Sistema de Monitoramento de Câmeras com Time-lapse

## OVERVIEW
Plataforma SaaS para monitoramento de câmeras IP distribuídas em múltiplas localidades, com captura automática de fotos e geração de time-lapses sob demanda.

**Modelo de Negócio:**
- Admin (você/equipe) gerencia todo o sistema
- Clientes pagam pelo serviço e veem apenas suas câmeras
- Multi-tenant: cada cliente vê apenas o que foi atribuído a ele

## STACK TÉCNICA
- Frontend: React + TypeScript + Tailwind CSS
- Backend: Node.js + Express + TypeScript
- Queue: Bull + Redis (processamento assíncrono)
- Video: Fluent-FFmpeg (geração de time-lapses)
- Database: Supabase PostgreSQL
- Storage: Supabase Storage (fotos e vídeos)
- Auth: Supabase Auth (JWT)
- Deploy: Render.com Pro ($85/mês) + Cron Job ($1/mês)

## ESTRUTURA DE DADOS (SUPABASE)

### Tabelas PostgreSQL
```sql
-- Clientes
CREATE TABLE clientes (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  nome TEXT NOT NULL,
  email TEXT UNIQUE NOT NULL,
  telefone TEXT,
  empresa TEXT,
  user_id UUID REFERENCES auth.users(id),
  status TEXT DEFAULT 'ativo',
  created_at TIMESTAMP DEFAULT NOW()
);

-- Localidades
CREATE TABLE localidades (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  nome TEXT NOT NULL,
  endereco TEXT,
  descricao TEXT,
  cliente_id UUID REFERENCES clientes(id),
  status TEXT DEFAULT 'ativo',
  created_at TIMESTAMP DEFAULT NOW()
);

-- Câmeras
CREATE TABLE cameras (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  localidade_id UUID REFERENCES localidades(id),
  nome TEXT NOT NULL,
  marca TEXT, -- reolink, intelbras, hikvision, outra
  modelo TEXT,
  hostname TEXT NOT NULL,
  porta_http INTEGER NOT NULL,
  porta_rtsp INTEGER,
  usuario TEXT NOT NULL,
  senha TEXT NOT NULL,
  intervalo_captura INTEGER DEFAULT 15,
  status TEXT DEFAULT 'online',
  ultima_captura TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Capturas
CREATE TABLE capturas (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  camera_id UUID REFERENCES cameras(id) ON DELETE CASCADE,
  imagem_url TEXT NOT NULL,
  imagem_path TEXT NOT NULL,
  tamanho_bytes INTEGER,
  capturado_em TIMESTAMP DEFAULT NOW(),
  INDEX idx_camera_data (camera_id, capturado_em DESC)
);

-- Time-lapses
CREATE TABLE timelapses (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  camera_id UUID REFERENCES cameras(id),
  solicitado_por UUID REFERENCES auth.users(id),
  nome TEXT,
  data_inicio DATE NOT NULL,
  data_fim DATE NOT NULL,
  fps INTEGER DEFAULT 30,
  status TEXT DEFAULT 'na_fila',
  progresso INTEGER DEFAULT 0,
  video_url TEXT,
  video_path TEXT,
  tamanho_bytes INTEGER,
  duracao_segundos INTEGER,
  total_frames INTEGER,
  erro_mensagem TEXT,
  created_at TIMESTAMP DEFAULT NOW(),
  completed_at TIMESTAMP
);
```

### RLS Policies
```sql
-- Clientes só veem suas localidades
ALTER TABLE localidades ENABLE ROW LEVEL SECURITY;
CREATE POLICY "clientes_suas_localidades" ON localidades FOR SELECT
USING (cliente_id IN (SELECT id FROM clientes WHERE user_id = auth.uid()));

-- Clientes só veem suas câmeras
ALTER TABLE cameras ENABLE ROW LEVEL SECURITY;
CREATE POLICY "clientes_suas_cameras" ON cameras FOR SELECT
USING (localidade_id IN (
  SELECT id FROM localidades WHERE cliente_id IN (
    SELECT id FROM clientes WHERE user_id = auth.uid()
  )
));

-- Admins veem tudo
CREATE POLICY "admins_tudo" ON localidades FOR ALL
USING (auth.jwt()->>'role' = 'admin');
```

### 6. CAPTURA AUTOMÁTICA (CRON JOB)
**Render Cron ($1/mês):**
```yaml
schedule: "*/5 * * * *" # A cada 5 minutos
```

**Lógica:**
```typescript
// Para cada câmera ativa
for (const camera of cameras) {
  // Verifica se está na hora de capturar
  const minutosDesdeUltimaCaptura = diff(now, camera.ultima_captura)
  
  if (minutosDesdeUltimaCaptura >= camera.intervalo_captura) {
    try {
      // Monta URL baseado na marca
      const url = camera.marca === 'reolink' 
        ? `http://${camera.hostname}:${camera.porta_http}/cgi-bin/api.cgi?cmd=Snap&channel=0&user=${camera.usuario}&password=${camera.senha}`
        : `http://${camera.hostname}:${camera.porta_http}/cgi-bin/snapshot.cgi?channel=1&user=${camera.usuario}&password=${camera.senha}`
      
      // Captura
      const response = await axios.get(url, { 
        responseType: 'arraybuffer',
        timeout: 10000 
      })
      
      // Gera path
      const path = `${ano}/${mes}/${dia}/camera-${camera.id}/${hora}-${min}-${seg}.jpg`
      
      // Upload Supabase
      await supabase.storage.from('capturas').upload(path, response.data)
      
      // Registra no banco
      await supabase.from('capturas').insert({
        camera_id: camera.id,
        imagem_url: getPublicUrl(path),
        imagem_path: path,
        tamanho_bytes: response.data.length
      })
      
      // Atualiza câmera
      await supabase.from('cameras').update({
        ultima_captura: now(),
        status: 'online'
      }).eq('id', camera.id)
      
    } catch (error) {
      // Incrementa contador de falhas
      // Se 3 falhas consecutivas: status='offline'
      // Log erro
    }
  }
}
```

**Worker (Render Pro - 8GB RAM):**
```typescript
async function processarTimelapse(job) {
  const { timelapseId, cameraId, dataInicio, dataFim, fps } = job.data
  
  try {
    // 1. Status processando
    await updateStatus(timelapseId, 'processando', 0)
    
    // 2. Buscar capturas
    const capturas = await buscarCapturas(cameraId, dataInicio, dataFim)
    
    // 3. Download imagens para /tmp (batch de 10 paralelo)
    const tempDir = `/tmp/timelapse-${timelapseId}`
    for (let i = 0; i < capturas.length; i += 10) {
      await Promise.all(batch.map(baixarImagem))
      await updateProgress(timelapseId, 10 + (i/capturas.length * 40))
    }
    
    // 4. FFmpeg gerar vídeo
    await ffmpeg()
      .input(`${tempDir}/frame-%06d.jpg`)
      .inputFPS(fps)
      .videoCodec('libx264')
      .size('1920x1080')
      .output(`${tempDir}/video.mp4`)
      .on('progress', p => updateProgress(timelapseId, 50 + p.percent * 0.4))
      .run()
    
    // 5. Upload Supabase
    const videoBuffer = await fs.readFile(`${tempDir}/video.mp4`)
    const path = `timelapses/timelapse-${timelapseId}/video.mp4`
    await supabase.storage.from('timelapses').upload(path, videoBuffer)
    
    // 6. Update status pronto
    await supabase.from('timelapses').update({
      status: 'pronto',
      progresso: 100,
      video_url: getPublicUrl(path),
      completed_at: now()
    }).eq('id', timelapseId)
    
    // 7. Limpar /tmp
    await fs.rm(tempDir, { recursive: true })
    
  } catch (error) {
    await updateStatus(timelapseId, 'erro', 0, error.message)
  }
}
```
